public class Automovil {
    String marca;
    String subMarca;
    int modelo;
    Color color;

    public Automovil() {
        
    }
    
    public Automovil(String marca, String subMarca, int modelo, Color color) {
        this.marca = marca;
        this.subMarca = subMarca;
        this.modelo = modelo;
        this.color = color;
    }
    

    

}
